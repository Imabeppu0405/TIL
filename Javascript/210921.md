# シューティングゲームを作る(第12回まで)


## 当たり判定
```js
//円形同士の当たり判定
function checkHit1(x1, y1, r1, x2, y2, r2) {
  let a = (x1-x2)>>8;
  let b = (y1-y2)>>8;
  let r = r1+r2;
  return (r*r >= a*a + b*b);
  //ピタゴラスの定理
}

// 四角同士の当たり判定
function checkHit2(x1, y1, w1, h1, x2, y2, x2, y2, w2, h2) {
  let left1 = x1>>8;
  let left2 = x2>>8;
  let top1 = y1>>8;
  let top2 = y2>>8;
  return (left1+w1 >= left2 && left1 <= left2+ w2 && top1+h1 >= top2 && top1 <= top2+h2)
}
```
